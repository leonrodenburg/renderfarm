225 DLL_API RFMath::Matrix RFMath::Inverse(const Matrix& matrix)
226 {
227  Matrix result;
228 
229  // Compute determinant of upper-left 3x3 matrix
230  float cofactor0 = matrix[5] * matrix[10] - matrix[6] * matrix[9];
231  float cofactor4 = matrix[2] * matrix[9] - matrix[1] * matrix[10];
232  float cofactor8 = matrix[1] * matrix[6] - matrix[2] * matrix[5];
233  float det = matrix[0] * cofactor0 + matrix[4] * cofactor4 + matrix[8] * cofactor8;
234 
235  // If the determinant is zero (and the matrix is not invertible)
236  if(RFMathIsZero(det))
237  {
238 #ifdef DEBUG
239  RFCore::Logger::GetLogger()->Log(RFCore::Logger::WARNING_TYPE, "Matrix has no inverse");
240 #endif
241 
242  return result;
243  }
244 
245  // Create adjunct matrix and multiply by 1 / determinant to get upper 3x3
246  float invDet = 1.0f / det;
247 
248  result._elements[0] = invDet * cofactor0;
249  result._elements[1] = invDet * cofactor4;
250  result._elements[2] = invDet * cofactor8;
251 
252  result._elements[4] = invDet * (matrix[6] * matrix[8] - matrix[4] * matrix[10]);
253  result._elements[5] = invDet * (matrix[0] * matrix[10] - matrix[2] * matrix[8]);
254  result._elements[6] = invDet * (matrix[2] * matrix[4] - matrix[0] * matrix[6]);
255 
256  result._elements[8] = invDet * (matrix[4] * matrix[9] - matrix[5] * matrix[8]);
257  result._elements[9] = invDet * (matrix[1] * matrix[8] - matrix[0] * matrix[9]);
258  result._elements[10] = invDet * (matrix[0] * matrix[5] - matrix[1] * matrix[4]);
259 
260  result._elements[12] = -result[0] * matrix[12] - result[4] * matrix[13] - result[8] * matrix[14];
261  result._elements[13] = -result[1] * matrix[12] - result[5] * matrix[13] - result[9] * matrix[14];
262  result._elements[14] = -result[2] * matrix[12] - result[6] * matrix[13] - result[10] * matrix[14];
263 
264  result.Clean();
265 
266  return result;
267 }
